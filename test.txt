//C0言語インタプリタは次のようなソースコードを読み取って、プログラムを実行します

//関数の外でグローバル変数を宣言できます
//変数として使用できるデータ型はint型とboolean型です
int a = 20;
boolean bool = true;

int main() {

	//複合文の先頭で、ローカル変数を宣言できます
	int i = 5;

	i = i + 10;

	//標準関数printを使って、変数や定数を表示できます
	print("i = %d%n", i);
	print("%d%n", a);
	print("%b%n", bool);
	
	return 0;
}
