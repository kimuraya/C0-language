//C0言語インタプリタは次のようなソースコードを読み取って、プログラムを実行します

int main() {
	
	int i = 1;
	
	//コメント
	i = i + 1; //代入式
	-10; //単項マイナス
	!i; //論理否定（整数なので、間違い）
	++i; //前置増分式 
	--i; //前置減分式
	i; //識別子
	10; //定数
	"文字列定数"; //文字列定数
	i++; //後置増分式
	i--; //後置減分式
	2 * 2; //乗算
	5 / 2; //除算
	
	i = add(i); //関数呼び出し
	
	i = (1 + 1) / 2; //括弧式
	
	//関係式
	if(1 < i) {
		i++;
	} else if (10 > i) {
		i++;
	} else if (12 <= i) {
		i++;
	} else if (12 >= i) {
		i++;
	}
	
	//同等式
	if(1 == i) {
		i++;
	} else if (10 != i) {
		i++;
	}
	
	//論理式
	if(1 && i) {
		i++;
	} else if (10 || i) {
		i++;
	}
	
	return i;
}

int add(int i) {
	i = i + 1;
	return i;
}
